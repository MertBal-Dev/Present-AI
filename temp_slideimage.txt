п»ҝimport React, { useState, useEffect, useMemo, useRef } from 'react';
import { Code, RefreshCw } from '../common/Icons'; 

function SlideImage({ keywords, onImageChange }) {
  const [imageUrl, setImageUrl] = useState(null);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState(false);
  const [page, setPage] = useState(1);
  const [retryCount, setRetryCount] = useState(0);
  
  // Kritik: Her sorgu iГғВ§in sadece BГ„В°R KEZ arama yap
  const fetchedQueries = useRef(new Set());
  const isFetching = useRef(false);
  const shouldRefresh = useRef(false);
  
  const API_BASE_URL = useMemo(() => (
    process.env.NODE_ENV === 'production'
      ? process.env.REACT_APP_API_BASE_URL
      : 'http://localhost:5001/api'
  ), []);

  // GГғВ¶rsel URL'ini parent'a gГғВ¶nder (sadece deГ„ЕёiГ…ЕёtiГ„Еёinde)
  useEffect(() => {
    if (imageUrl && onImageChange) {
      onImageChange(imageUrl);
    }
  }, [imageUrl]); // onImageChange'i dependency'den ГғВ§Г„ВұkardГ„Вұk

  useEffect(() => {
    const query = keywords?.query;
    
    // Query yoksa durdur
    if (!query) {
      setIsLoading(false);
      return;
    }

    // AynГ„Вұ query iГғВ§in zaten arama yapГ„ВұldГ„Вұysa ve manuel yenileme yoksa durdur
    const queryKey = `${query}-${page}`;
    if (fetchedQueries.current.has(queryKey) && !shouldRefresh.current) {
      console.log(`ГўЕ“вҖҰ [Cache] "${query}" iГғВ§in zaten arama yapГ„ВұlmГ„ВұГ…Её, atlanГ„Вұyor.`);
      return;
    }

    // Г…Вһu anda arama yapГ„ВұlГ„Вұyorsa ГғВ§ift arama yapma
    if (isFetching.current) {
      console.log(`ГўВҸВі [Skip] "${query}" iГғВ§in zaten arama yapГ„ВұlГ„Вұyor...`);
      return;
    }

    const fetchImage = async () => {
      isFetching.current = true;
      shouldRefresh.current = false;
      setIsLoading(true);
      setError(false);
      
      console.log(`ДҹЕёвҖқВҚ [Fetch] "${query}" iГғВ§in gГғВ¶rsel aranГ„Вұyor (page: ${page})...`);
      
      try {
        const response = await fetch(`${API_BASE_URL}/search-image`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            query: query,
            page: page,
            nonce: Date.now(),
          }),
        });

        if (!response.ok) throw new Error('Image fetch failed');

        const data = await response.json();
        if (data.imageUrl) {
          setImageUrl(data.imageUrl);
          fetchedQueries.current.add(queryKey);
          console.log(`ГўЕ“вҖҰ [Success] "${query}" iГғВ§in gГғВ¶rsel bulundu: ${data.imageUrl}`);
        } else {
          setError(true);
          console.log(`ГўВқЕ’ [Error] "${query}" iГғВ§in gГғВ¶rsel bulunamadГ„Вұ`);
        }
      } catch (err) {
        console.error(`ГўВқЕ’ [Error] "${query}" arama hatasГ„Вұ:`, err);
        setError(true);
      } finally {
        setIsLoading(false);
        isFetching.current = false;
      }
    };

    fetchImage();
  }, [keywords?.query, page, API_BASE_URL]);

  // Manuel yenileme butonu
  const handleRefresh = () => {
    console.log(`ДҹЕёвҖқвҖһ [Refresh] Manuel yenileme baГ…ЕёlatГ„ВұlГ„Вұyor...`);
    shouldRefresh.current = true;
    
    if (retryCount < 2) {
      // AynГ„Вұ sayfada yeniden dene
      setRetryCount(prev => prev + 1);
      const queryKey = `${keywords?.query}-${page}`;
      fetchedQueries.current.delete(queryKey);
      // Force re-fetch
      setImageUrl(null);
    } else {
      // 3. denemede sonraki sayfaya geГғВ§
      setPage(prevPage => prevPage + 1);
      setRetryCount(0);
    }
  };

  // Sorgu deГ„ЕёiГ…ЕёtiГ„Еёinde sГ„ВұfГ„Вұrla
  const prevQuery = useRef(null);
  useEffect(() => {
    if (keywords?.query !== prevQuery.current) {
      console.log(`ДҹЕёвҖ вҖў [New Query] Sorgu deГ„ЕёiГ…Еёti: "${prevQuery.current}" ГўвҖ вҖҷ "${keywords?.query}"`);
      prevQuery.current = keywords?.query;
      setPage(1);
      setRetryCount(0);
      setError(false);
      setImageUrl(null);
    }
  }, [keywords?.query]);

  if (isLoading) {
    return (
      <div className="w-full h-full flex items-center justify-center bg-gradient-to-br from-gray-100 to-gray-200">
        <div className="flex flex-col items-center space-y-3">
          <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-indigo-600"></div>
          <p className="text-sm text-gray-500">Kaliteli gГғВ¶rsel aranГ„Вұyor...</p>
        </div>
      </div>
    );
  }

  if (error || !imageUrl) {
    return (
      <div className="w-full h-full flex items-center justify-center bg-gradient-to-br from-indigo-100 via-purple-50 to-pink-100">
        <div className="text-center p-8">
          <div className="w-32 h-32 mx-auto mb-4 rounded-2xl bg-gradient-to-br from-indigo-400 to-purple-500 flex items-center justify-center">
            <Code className="w-16 h-16 text-white" />
          </div>
          <p className="text-gray-500 text-sm mb-4">{keywords?.query || 'GГғВ¶rsel bulunamadГ„Вұ'}</p>
          <button 
            onClick={handleRefresh}
            className="px-4 py-2 bg-indigo-500 text-white rounded-lg hover:bg-indigo-600 transition-colors inline-flex items-center gap-2"
          >
            <RefreshCw className="w-4 h-4" />
            Tekrar Dene
          </button>
        </div>
      </div>
    );
  }

  return (
    <div className="w-full h-full relative overflow-hidden group">
      <img
        src={imageUrl}
        alt={keywords?.query || 'Slide image'}
        className="w-full h-full object-cover transition-transform duration-300 group-hover:scale-105"
        onError={() => setError(true)}
      />
      <div className="absolute inset-0 bg-black/20 opacity-0 group-hover:opacity-100 transition-opacity duration-300 flex items-center justify-center">
        <button 
          onClick={handleRefresh}
          className="p-3 bg-white/90 rounded-full text-gray-800 hover:bg-white hover:scale-110 transform transition-all shadow-lg"
          title="BaГ…Еёka GГғВ¶rsel Bul"
        >
          <RefreshCw className="w-6 h-6" />
        </button>
      </div>
      <div className="absolute bottom-2 right-2 bg-black/70 text-white text-xs px-3 py-1.5 rounded-md backdrop-blur-sm">
        {keywords?.query}
      </div>
      <div className="absolute top-2 right-2 bg-green-500/80 text-white text-xs px-2 py-1 rounded-md backdrop-blur-sm">
        HD
      </div>
    </div>
  );
}

export default SlideImage;
